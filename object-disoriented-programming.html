            # Define
class Person(object):
    def __init__(self, name):
        self.name = name

    def introduce(self):
        print("Hello, my name is %s" % self.name)

# Run
myself = Person('Johannes')
myself.introduce()

# Same as
myself = Person('Johannes')
Person.introduce(myself)
